local Player = game.Players.LocalPlayer
local Character = Player.Character or Player.CharacterAdded:Wait()
local Root = Character:WaitForChild("HumanoidRootPart")

_G.Active = true

spawn(function()
    while _G.Active do
        task.wait(0.1)
        pcall(function()
            -- Find enemies in Workspace
            for _, v in pairs(game:GetService("Workspace"):GetDescendants()) do
                if v:IsA("Model") and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 and v.Name ~= Player.Name then
                    repeat
                        task.wait()
                        -- Forced TP above enemy head
                        Root.CFrame = v.HumanoidRootPart.CFrame * CFrame.new(0, 6, 0)
                        
                        -- Infinite Attack
                        game:GetService("VirtualUser"):Button1Down(Vector2.new(0,0), workspace.CurrentCamera.CFrame)
                    until v.Humanoid.Health <= 0 or not _G.Active
                end
            end
        end)
    end
end)
